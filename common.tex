% SWITCH Statement
\newcommand{\xcase}[2]{#1 #2} % Dummy, so \renewcommand has something to overwrite...

% Repeat a String N Times
\makeatletter%
\newcount\x@repeat@count%
\newcommand{\xrepeat}[2]{%%
  \begingroup%
  \x@repeat@count=\z@%
  \@whilenum\x@repeat@count<#1\do{#2\advance\x@repeat@count\@ne}%%
  \endgroup%
}%
\makeatother%
