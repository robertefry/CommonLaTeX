
\usepackage{newpxtext}
\usepackage{setspace}

\makeatletter
\edef\@subtitle{}
\NewDocumentCommand{\subtitle}{m}%
{%
    \def\@subtitle{#1}%
}%
\makeatother

\makeatletter
\newif\if@version@set\@version@setfalse
\edef\@version{1.0.0}
\NewDocumentCommand{\version}{mO{0}O{0}D<>{}}%
{%
    \edef\@version{#1.#2.#3\ifx\relax#4\relax\else-\fi#4}%
    \@version@settrue%
}%
\makeatother

\makeatletter
\edef\@supervisors{}
\NewDocumentCommand{\supervisor}{m}%
{%
    \ifx\@supervisors\empty%
        \def\@supervisors{#1}%
    \else%
        \g@addto@macro\@supervisors{, #1}%
    \fi%
}%
\makeatother

\makeatletter\renewcommand\maketitle{
\begin{titlepage}\centering

    \hrule width \hsize height 0.4mm
    \kern 0.8mm
    \hrule width \hsize height 0.2mm
    \kern -2.0mm
    {
        \vspace*{5mm}
        \Huge\sc\textbf\@title
        \ifx\@subtitle\empty\else{\par\vspace*{2mm}\Large\sc\@subtitle}\fi
        \vspace*{5mm}
    }
    \hrule width \hsize height 0.2mm
    \kern 0.8mm
    \hrule width \hsize height 0.4mm

    \kern 1cm

    \begin{spacing}{1.5}
    {
        \sc
        \begin{minipage}[t]{0.55\textwidth}\raggedright
            Author: \@author\\
            \ifx\@supervisors\empty\else{Supervisors: \@supervisors\\}\fi
        \end{minipage}
        \hfill
        \begin{minipage}[t]{0.4\textwidth}\raggedleft
            \today\\
            \if@version@set{Version: \@version\\}\fi
        \end{minipage}
    }
    \end{spacing}

\end{titlepage}
}\makeatother
